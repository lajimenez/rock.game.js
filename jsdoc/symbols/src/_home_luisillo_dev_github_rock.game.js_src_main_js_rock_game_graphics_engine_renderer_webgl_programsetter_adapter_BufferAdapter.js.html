  3 /**
  4  * Adapter for store values in WebGL buffer.
  5  *
  6  * @constructor
  7  * @extends rock.game.graphics.engine.renderer.webgl.programsetter.adapter.Adapter
  8  *
  9  * @author Luis Alberto Jim√©nez
 10  */
 11 rock.game.graphics.engine.renderer.webgl.programsetter.adapter.BufferAdapter = function (glContext, values, typeArray, bufferType) {
 12     rock.super_(this, [glContext]);
 13     this.values = values;
 14     this.typeArray = typeArray;
 15     this.bufferType = bufferType;
 16 
 17     this.buffer = null;
 18     this.array = null;
 19 
 20     // Handle context events
 21     var app = glContext.getApplication();
 22     app.addEventListener(rock.constants.ROCK_EVENT_CONTEXT_RESTORED, rock.createEventHandler(this, this.onContextRestored));
 23 };

 25 rock.extends_(rock.game.graphics.engine.renderer.webgl.programsetter.adapter.BufferAdapter,
 26     rock.game.graphics.engine.renderer.webgl.programsetter.adapter.Adapter);

 51 /**
 52  * @private
 53  */
 54 rock.game.graphics.engine.renderer.webgl.programsetter.adapter.BufferAdapter.prototype.getGlBufferType = function () {
 55     var gl = this.glContext.getHTMLContext();
 56     var bufferType = this.bufferType;
 57     var glBufferType = null;
 58 
 59     switch(bufferType) {
 60         case rock.game.graphics.engine.renderer.webgl.programsetter.adapter.BufferAdapter.ARRAY_BUFFER:
 61             glBufferType = gl.ARRAY_BUFFER;
 62             break;
 63         case rock.game.graphics.engine.renderer.webgl.programsetter.adapter.BufferAdapter.ELEMENT_ARRAY_BUFFER:
 64             glBufferType = gl.ELEMENT_ARRAY_BUFFER;
 65             break;
 66         default:
 67             ;
 68     }
 69 
 70     return glBufferType;
 71 };

 73 /**
 74  * @see rock.game.graphics.engine.renderer.webgl.programsetter.adapter.Adapter#build
 75  * @override
 76  */
 77 rock.game.graphics.engine.renderer.webgl.programsetter.adapter.BufferAdapter.prototype.build = function () {
 78     var gl = this.glContext.getHTMLContext();
 79     var bufferType = this.getGlBufferType();
 80     var buffer = gl.createBuffer();
 81     var array =  new this.typeArray(this.values);
 82     gl.bindBuffer(bufferType, buffer);
 83     gl.bufferData(bufferType, array, gl.STATIC_DRAW);
 84 
 85     this.buffer = buffer;
 86     this.array = array;
 87 
 88     gl.bindBuffer(bufferType, null);
 89 };
 90 
 91 /**
 92  * @see rock.game.graphics.engine.renderer.webgl.programsetter.adapter.Adapter#release
 93  * @override
 94  */
 95 rock.game.graphics.engine.renderer.webgl.programsetter.adapter.BufferAdapter.prototype.release = function () {
 96     var gl = this.glContext.getHTMLContext();
 97     gl.deleteBuffer(this.buffer);
 98     this.array = null;
 99 };
