 11 rock.game.graphics.engine.renderer.webgl.programsetter.adapter.TextureAdapter = function (glContext, texture) {

 33 /**

 35  *
 36  * @private

 38 rock.game.graphics.engine.renderer.webgl.programsetter.adapter.TextureAdapter.prototype.restore  = function () {

 41 
 42 /**
 43  * @see rock.game.graphics.engine.renderer.webgl.programsetter.adapter.Adapter#build
 44  * @override
 45  */
 46 rock.game.graphics.engine.renderer.webgl.programsetter.adapter.TextureAdapter.prototype.build  = function () {
 47     var gl = this.glContext.getHTMLContext();
 48     var texture = this.texture;
 49     var textureImage = texture.getImage();
 50     this.glTexture = rock.util.WebGlUtils.populateTexture(gl, textureImage);
 51 };
 52 
 53 /**
 54  * @see rock.game.graphics.engine.renderer.webgl.programsetter.adapter.Adapter#release
 55  * @override
 56  */
 57 rock.game.graphics.engine.renderer.webgl.programsetter.adapter.TextureAdapter.prototype.release  = function () {
 58     var gl = this.glContext.getHTMLContext();
 59     var glTexture = this.glTexture;
 60 
 61     if (glTexture != null) {
 62         gl.deleteTexture(glTexture);
 63         this.glTexture = null;
 64     }
 65 };

 70 rock.game.graphics.engine.renderer.webgl.programsetter.adapter.TextureAdapter.prototype.getGlTexture = function() {
