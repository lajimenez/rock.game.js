 16 rock.game.graphics.engine.renderer.webgl.RendererContext = function (id, glContext) {
 17     rock.super_(this, [id]);
 18     this.glContext = glContext;
 19 };
 20 
 21 rock.extends_(rock.game.graphics.engine.renderer.webgl.RendererContext,
 22     rock.game.graphics.engine.renderer.RendererContext);
 23 
 24 rock.game.graphics.engine.renderer.webgl.RendererContext.prototype.setUp = function () {
 25     var gl = this.glContext.getHTMLContext();
 26     gl.enable(gl.DEPTH_TEST);
 27 
 28     gl.enable(gl.BLEND);
 29     gl.blendEquation(gl.FUNC_ADD);
 30     gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA);
 31 
 32     // gl.frontFace(gl.CCW); // CW - CCW
 33     gl.enable(gl.CULL_FACE);
 34     gl.cullFace(gl.BACK); // FRONT - BACK - FRONT_AND_BACK
 35 };
 36 
 37 rock.game.graphics.engine.renderer.webgl.RendererContext.prototype.clear = function () {
 38     var gl = this.glContext.getHTMLContext();
 39     var color = this.clearColor;
 40     gl.clearColor(color.getNormalizedRed(), color.getNormalizedGreen(), color
 41         .getNormalizedBlue(), color.getNormalizedAlpha());
 42     gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
 43 };
