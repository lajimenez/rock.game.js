 36 /**
 37  * Get a mesh containing all normals
 38  *
 39  * @returns {rock.game.graphics.model.mesh.Mesh} mesh containing all normals
 40  */
 41 rock.game.graphics.model.mesh.NormalMesh.prototype.getNormalsAsMesh = function () {
 42     // A more accurate implementation should get vertex and normals from 'indexes'
 43     // But then you get a lot of repeated elements (and there can not be more
 44     // normals than exists in array 'normals'... ). Well, you can save this with some
 45     // kind hash implementation but I don't think it's necessary...
 46     // The problem with this implementation is that if you have some vertex defined
 47     // but not used it will also appear... If you have used the rock.converter it could
 48     // happen if the mesh had wrong normals and the rock.converter has 'repaired' it
 49     // (but if the mesh was correct and you use rock.converter, the normal mesh will be ok :)
 50     var normalMesh = new rock.game.graphics.model.mesh.Mesh();
 51 
 52     var normals = this.normals;
 53     var length = normals.length;
 54     var vertices = this.vertices;
 55     var newVertices = [];
 56     var newIndexes = [];
 57     var i, newIndex = 0;
 58 
 59     for (i = 0; i < length; i = i + 3){
 60         newVertices.push(vertices[i], vertices[i +1], vertices[i + 2]);
 61         newVertices.push(vertices[i] + normals[i], vertices[i + 1] + normals[i + 1], vertices[i + 2] + normals[i + 2]);
 62 
 63         newIndexes.push(newIndex, newIndex + 1);
 64         newIndex += 2;
 65     }
 66 
 67     normalMesh.setVertices(newVertices);
 68     normalMesh.setIndexes(newIndexes);
 69     normalMesh.computeBBOX();
 70     return normalMesh;
 71 };
