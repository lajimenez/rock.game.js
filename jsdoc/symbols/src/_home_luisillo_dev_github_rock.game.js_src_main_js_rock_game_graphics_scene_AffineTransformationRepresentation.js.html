  3 /**
  4  * A representation that makes easy to apply affine transformations to its model.
  5  *
  6  * @constructor
  7  * @abstract
  8  * @extends rock.game.graphics.scene.Representation
  9  *
 10  * @author Luis Alberto Jim√©nez
 11  */
 12 rock.game.graphics.scene.AffineTransformationRepresentation = function (representationAttendant, object) {
 13     rock.super_(this, arguments);
 14 
 15     this.applyAffineTransformationsFromModelCenter = false;
 16 
 17     this.translation = new rock.geometry.Point3(0, 0, 0);
 18 
 19     this.rotationMode = rock.game.graphics.scene.AffineTransformationRepresentation.ROTATION_MODE_XYZ;
 20     this.rotationX = 0;
 21     this.rotationY = 0;
 22     this.rotationZ = 0;
 23 
 24     this.scaleX = 1;
 25     this.scaleY = 1;
 26     this.scaleZ = 1;
 27 };
 28 
 29 rock.extends_(rock.game.graphics.scene.AffineTransformationRepresentation, rock.game.graphics.scene.Representation);
 30 
 31 /**
 32  * In this mode, the rotations are applied in the next order: axis X, axis Y and axis Z.
 33  * @const
 34  */
 35 rock.game.graphics.scene.AffineTransformationRepresentation.ROTATION_MODE_XYZ = 'ROTATION_MODE_XYZ';
 36 
 37 rock.game.graphics.scene.AffineTransformationRepresentation.prototype.updateModelView = function () {
 38     this.setAffineTransformationValues();
 39     this.applyAffineTransformations();
 40 };
 41 
 42 /**
 43  *
 44  * @function
 45  */
 46 rock.game.graphics.scene.AffineTransformationRepresentation.prototype.setAffineTransformationValues = rock.abstract_;
 47 
 48 rock.game.graphics.scene.AffineTransformationRepresentation.prototype.applyAffineTransformations = function () {
 49     // Calculate modelMatrix
 50     var modelMatrix = this.matrix_ma;
 51     this.computeModelMatrix(modelMatrix);
 52 
 53     // Apply modelMatrix
 54     var renderables = this.renderables;
 55     var length = renderables.getLength();
 56     var i, renderable, renderableModelMatrix;
 57     for (i = 0; i < length; i++) {
 58         renderable = renderables.getValue(i);
 59         renderableModelMatrix = renderable.getModelMatrix();
 60         renderableModelMatrix.initFromMatrix(modelMatrix);
 61     }
 62 };
 63 
 64 rock.game.graphics.scene.AffineTransformationRepresentation.prototype.computeModelMatrix = function (modelMatrix) {
 65     modelMatrix.identity();
 66 
 67     // the order applied is inverse
 68     this.applyTranslation(modelMatrix);
 69     this.applyRotation(modelMatrix);
 70     this.applyScale(modelMatrix);
 71 
 72     this.moveModelCenterToOrigin(modelMatrix);
 73 };
 74 
 75 rock.game.graphics.scene.AffineTransformationRepresentation.prototype.applyTranslation = function (modelMatrix) {
 76     var translationPoint = this.translation;
 77     modelMatrix.translate(translationPoint.getX(), translationPoint.getY(), translationPoint.getZ());
 78 };
 79 
 80 rock.game.graphics.scene.AffineTransformationRepresentation.prototype.applyRotation = function (modelMatrix) {
 81     var rotationMode = this.rotationMode;
 82 
 83     if (rotationMode == rock.game.graphics.scene.AffineTransformationRepresentation.ROTATION_MODE_XYZ) {
 84         modelMatrix.rotateX(this.rotationX);
 85         modelMatrix.rotateY(this.rotationY);
 86         modelMatrix.rotateZ(this.rotationZ);
 87     } else {
 88         // TODO: throw exception
 89     }
 90 };
 91 
 92 rock.game.graphics.scene.AffineTransformationRepresentation.prototype.applyScale = function (modelMatrix) {
 93     modelMatrix.scale(this.scaleX, this.scaleY, this.scaleZ);
 94 };
 95 
 96 rock.game.graphics.scene.AffineTransformationRepresentation.prototype.moveModelCenterToOrigin = function (modelMatrix) {
 97     if (!this.applyAffineTransformationsFromModelCenter) {
 98         return;
 99     }
100 
101     var modelCenter = this.model.getCenter();
102     modelMatrix.translate(-modelCenter.getX(), -modelCenter.getY(), -modelCenter.getZ());
103 };
