 68 rock.scene.node.ModelNode.prototype.getBBOX = function () {
 69     var modelCenter;
 70     var model = this.representation.getModel();
 71     var transformedBBOX = model.getBBOX();
 72     if (this.applyAffineTransformationsFromModelCenter) {
 73         modelCenter = model.getCenter();
 74         transformedBBOX = transformedBBOX.getTranslatedBBOX(-modelCenter.getX(), -modelCenter.getY(), -modelCenter.getZ());
 75     }
 76 
 77     var position = this.position;
 78     transformedBBOX = transformedBBOX.getRotatedBBOX(this.rotationX, this.rotationY, this.rotationZ);
 79     transformedBBOX = transformedBBOX.getScaledBBOX(this.scaleX, this.scaleY, this.scaleZ);
 80     transformedBBOX = transformedBBOX.getTranslatedBBOX(position.getX(), position.getY(), position.getZ());
 81 
 82     return transformedBBOX;
 83 };
