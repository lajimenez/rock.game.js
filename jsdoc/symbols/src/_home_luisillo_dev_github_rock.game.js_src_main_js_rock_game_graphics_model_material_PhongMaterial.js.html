 11 rock.game.graphics.model.material.PhongMaterial = function () {
 12     rock.super_(this);
 13     // {Array.4}
 14     this.ambient = new rock.graphics.Color(0, 0, 0);
 15     this.diffuse = new rock.graphics.Color(0, 0, 0);
 16     this.specular = new rock.graphics.Color(0, 0, 0);
 17     this.shininess = 1;
 18     this.alpha = 1;
 19 };

 23 rock.game.graphics.model.material.PhongMaterial.prototype.loadFromJSON = function (JSON) {
 24     var JSONPhongColor = JSON.phongColor;
 25     var ambient = JSONPhongColor.ambient;
 26     var diffuse = JSONPhongColor.diffuse;
 27     var specular = JSONPhongColor.specular;
 28     var shininess = JSONPhongColor.shininess;
 29     var alpha = JSONPhongColor.alpha;
 30     this.setAmbient(ambient[0] * 255, ambient[1] * 255, ambient[2] * 255);
 31     this.setDiffuse(diffuse[0] * 255, diffuse[1] * 255, diffuse[2] * 255);
 32     this.setSpecular(specular[0] * 255, specular[1] * 255, specular[2] * 255);
 33     this.setShininess(shininess);
 34     this.setAlpha(alpha);
 35 };

108     var clone = new rock.game.graphics.model.material.PhongMaterial();
109     var ambient = this.ambient;
110     var diffuse = this.diffuse;
111     var specular = this.ambient;
112

114     clone.setDiffuse(diffuse.getRed(), diffuse.getGreen(), diffuse.getBlue());
115     clone.setSpecular(specular.getRed(), specular.getGreen(), specular.getBlue());
116     clone.setShininess(this.getShininess());
117
