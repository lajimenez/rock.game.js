  3 /**
  4  * All world objects that you want to draw must have a representation.
  5  * A representation can be understood as a controller that is responsible to prepare a model
  6  * according to its world object.
  7  * A representation must know how a world object must be drawn (and must be the only one :)
  8  *
  9  * @param {rock.game.engine.RepresentationAttendant} representationAttendant
 10  * @param {rock.game.universe.Object} object
 11  *
 12  * @constructor
 13  * @abstract
 14  *
 15  * @author Luis Alberto Jim√©nez
 16  */
 17 rock.game.graphics.scene.Representation = function (representationAttendant, object) {
 18     this.representationAttendant = representationAttendant;
 19     this.object = object;
 20     this.model = null;
 21     this.renderables = new rock.js.Array();
 22 
 23     this.matrix_ma = new rock.geometry.Matrix4();
 24 
 25     this.updateModel();
 26     this.createRenderables();
 27 };

 70 /**
 71  * This function update the model matrix from the object position
 72  */
 73 rock.game.graphics.scene.Representation.prototype.updateRenderablesModelMatrixFromPosition = function () {
 74     var position = this.object.getPosition();
 75     var renderables = this.renderables;
 76     var positionModelMatrix = this.matrix_ma;
 77     positionModelMatrix.identity();
 78     positionModelMatrix.translate(position.getX(), position.getY(), position.getZ());
 79 
 80     var i, renderable, renderableModelMatrix, length = renderables.getLength();
 81     for (i = 0; i < length; i++) {
 82         renderable = renderables.getValue(0);
 83         renderableModelMatrix = renderable.getModelMatrix();
 84         renderableModelMatrix.initFromMatrix(positionModelMatrix);
 85     }
 86 };
