  3 /**
  4  * Model renderer for using with WebGL.
  5  * This is the class that you should implement when creating a new renderer.
  6  *
  7  * @param {String} lightingType
  8  *         the lighting type
  9  * @param {String} modelType
 10  *         the model type
 11  * @param {rock.graphics.engine.webgl.WebGLContext} glContext
 12  *         the WebGL context
 13  * @param {Number} drawingMode
 14  *         gl drawing model
 15  * @param {String} vertexShaderSrc
 16  *         source for the vertex shader
 17  * @param {String} fragmentShaderSrc
 18  *         source for the fragment shader
 19  * @param {rock.game.graphics.engine.renderer.webgl.programsetter.setter.lighting.LightingSetter} lightingSetter
 20  *          lighting setter
 21  * @param {rock.game.graphics.engine.renderer.webgl.programsetter.setter.model.ModelSetter} modelSetter
 22  *          model setter
 23  * @param {Boolean} [provisioningBuffers]
 24  *          if true, caching will also create buffers
 25  * @param {Boolean} [provisioningTextures]
 26  *          if true, caching will also create textures
 27  *
 28  * @constructor
 29  * @extends rock.game.graphics.engine.renderer.ModelRenderer
 30  *
 31  * @author Luis Alberto Jim√©nez
 32  */
 33 rock.game.graphics.engine.renderer.webgl.ModelRenderer = function (lightingType, modelType,
 34                                                                    glContext, drawingMode,
 35                                                                    vertexShaderSrc, fragmentShaderSrc,
 36                                                                    lightingSetter, modelSetter,
 37                                                                    provisioningBuffers, provisioningTextures) {
 38     rock.super_(this, [lightingType, modelType]);
 39     this.glContext = glContext;
 40 
 41     this.matricesSetter =
 42         new rock.game.graphics.engine.renderer.webgl.programsetter.setter.model.MatricesSetter(glContext.getHTMLContext());
 43     this.lightingSetter = lightingSetter;
 44     this.modelSetter = modelSetter;
 45 
 46     this.matricesSettable = new rock.game.graphics.engine.renderer.webgl.programsetter.settable.MatricesSettable();
 47     this.lightingSettable = new rock.game.graphics.engine.renderer.webgl.programsetter.settable.LightingSettable();
 48     this.modelSettable = new rock.game.graphics.engine.renderer.webgl.programsetter.settable.ModelSettable();
 49 
 50     this.drawingMode = drawingMode;
 51 
 52     var attributes = this.getAttributeNames();
 53     var uniforms = this.getUniformNames();
 54     var textures = this.getTextureNames();
 55     this.program = new rock.graphics.engine.renderer.webgl.program.CacheProgram(
 56         glContext, vertexShaderSrc, fragmentShaderSrc, attributes, uniforms, textures,
 57         provisioningBuffers, provisioningTextures);
 58 
 59     // this properties are to avoid memory allocation
 60     this.modelViewMatrix_ma = new rock.geometry.Matrix4();
 61     this.normalMatrix_ma = new rock.geometry.Matrix4();
 62 };

 92 /**
 93  * @see rock.game.graphics.engine.renderer.ModelRenderer#render
 94  * @override
 95  */
 96 rock.game.graphics.engine.renderer.webgl.ModelRenderer.prototype.render = function (lighting, camera, renderable) {
 97     var gl = this.glContext.getHTMLContext();
 98     var program = this.program;
 99 
100     program.useProgram();
101 
102     this.setCommonProgramValues(lighting, camera, renderable);
103 
104     program.enableAttributes();
105 
106     var model = renderable.getModel();
107     var modelAdapter = renderable.getModelAdapter();
108     this.setProgramValuesModel(model, modelAdapter);
109 
110     gl.drawElements(this.drawingMode, model.getMeshCount(), gl.UNSIGNED_SHORT, 0);
111 
112     program.disableAttributes();
113 
114     gl.bindBuffer(gl.ARRAY_BUFFER, null);
115     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, null);
116 };

118 rock.game.graphics.engine.renderer.webgl.ModelRenderer.prototype.setCommonProgramValues = function (lighting, camera, renderable) {
119     this.setProgramValuesMatrices(renderable, camera);
120     this.setProgramValuesLighting(lighting, camera);
121 };
122 
123 rock.game.graphics.engine.renderer.webgl.ModelRenderer.prototype.setProgramValuesMatrices = function (renderable, camera) {
124     var matricesSettable = this.matricesSettable;
125     var viewMatrix = camera.getLookAtViewMatrix();
126     var projectionMatrix = camera.getPerspectiveProjectionMatrix();
127     var modelMatrix = renderable.getModelMatrix();
128 
129     var modelViewMatrix = this.modelViewMatrix_ma;
130     var normalMatrix = this.normalMatrix_ma;
131 
132     modelViewMatrix.initFromMatrix(viewMatrix);
133     modelViewMatrix.multiply(modelMatrix);
134 
135     normalMatrix.initFromMatrix(modelViewMatrix);
136     normalMatrix.invert();
137     normalMatrix.transpose();
138     normalMatrix.removeTranslation();
139 
140     matricesSettable.setModelViewMatrix(modelViewMatrix);
141     matricesSettable.setProjectionMatrix(projectionMatrix);
142     matricesSettable.setNormalMatrix(normalMatrix);
143 
144     this.matricesSetter.setValues(this.program, matricesSettable);
145 };
146 
147 rock.game.graphics.engine.renderer.webgl.ModelRenderer.prototype.setProgramValuesLighting = function (lighting, camera) {
148     var lightingSettable = this.lightingSettable;
149     lightingSettable.setLighting(lighting);
150     lightingSettable.setCamera(camera);
151     this.lightingSetter.setValues(this.program, lightingSettable);
152 };
153 
154 rock.game.graphics.engine.renderer.webgl.ModelRenderer.prototype.setProgramValuesModel = function (model, modelAdapter) {
155     var modelSettable = this.modelSettable;
156     modelSettable.setModel(model);
157     modelSettable.setAdapter(modelAdapter);
158     this.modelSetter.setValues(this.program, modelSettable);
159 };
