 14 rock.game.graphics.engine.renderer.webgl.programsetter.setter.model.MatricesSetter = function (glContext) {
 15     this.UNIFORM_MODEL_VIEW_MATRIX = 'u_modelViewMatrix';
 16     this.UNIFORM_PROJECTION_MATRIX = 'u_projectionMatrix';

 37 rock.game.graphics.engine.renderer.webgl.programsetter.setter.model.MatricesSetter.prototype.setValues = function (program, settable) {
 38     var gl = this.glContext;
 39 
 40     var modelViewMatrix = settable.getModelViewMatrix();
 41     var projectionMatrix = settable.getProjectionMatrix();
 42     var normalMatrix = settable.getNormalMatrix();
 43 
 44     gl.uniformMatrix4fv(
 45         program.getUniformLocation(this.UNIFORM_MODEL_VIEW_MATRIX),
 46         gl.FALSE,
 47         modelViewMatrix.getMatrixAsArray()
 48     );
 49 
 50     gl.uniformMatrix4fv(
 51         program.getUniformLocation(this.UNIFORM_PROJECTION_MATRIX),
 52         gl.FALSE,
 53         projectionMatrix.getMatrixAsArray()
 54     );
 55 
 56     var normalMatrixUniformLocator = program.getUniformLocation(this.UNIFORM_NORMAL_MATRIX);
 57 
 58     if (normalMatrixUniformLocator != null) {
 59         gl.uniformMatrix3fv(
 60             normalMatrixUniformLocator,
 61             gl.FALSE,
 62             normalMatrix.getAsMatrix3().getMatrixAsArray()
 63         );
 64     }
 65 };
