 11 rock.game.graphics.model.material.PhongTextureMaterial = function () {
 12     rock.super_(this);
 13     this.ambientTexture = null;
 14     this.diffuseTexture = null;
 15     this.specularTexture = null;
 16 };
 17 
 18 rock.extends_(rock.game.graphics.model.material.PhongTextureMaterial, rock.game.graphics.model.material.PhongMaterial);
 19 
 20 rock.game.graphics.model.material.PhongTextureMaterial.prototype.loadFromJSON = function (JSON, repository) {
 21     rock.super_method(this, rock.game.graphics.model.material.PhongTextureMaterial, 'loadFromJSON', arguments);
 22     var JSONPhongTexture = JSON.phongTexture;
 23     var image;
 24     if (!rock.util.JsUtils.isNullOrUndefined(JSONPhongTexture.ambient)) {
 25         image = repository.getTextureImage(JSONPhongTexture.ambient);
 26         this.ambientTexture = new rock.game.graphics.model.material.Texture(image);
 27     }
 28 
 29     if (!rock.util.JsUtils.isNullOrUndefined(JSONPhongTexture.diffuse)) {
 30         image = repository.getTextureImage(JSONPhongTexture.diffuse);
 31         this.diffuseTexture = new rock.game.graphics.model.material.Texture(image);
 32     }
 33 
 34     if (!rock.util.JsUtils.isNullOrUndefined(JSONPhongTexture.specular)) {
 35         image = repository.getTextureImage(JSONPhongTexture.specular);
 36         this.specularTexture = new rock.game.graphics.model.material.Texture(image);
 37     }
 38 };
