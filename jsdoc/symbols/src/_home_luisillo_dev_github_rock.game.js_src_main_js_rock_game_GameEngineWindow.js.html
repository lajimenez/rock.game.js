 19 rock.game.GameEngineWindow = function (windowSystem, gameDirector, gameEngine) {
 20     rock.super_(this, [windowSystem, gameDirector]);
 21     this.gameEngine = gameEngine;
 22     this.gameEngine.setWindow(this);
 23 };
 24 
 25 rock.extends_(rock.game.GameEngineWindow, rock.game.GameWindow);
 26 
 27 /**
 28  * @see rock.graphics.engine.IDrawable#draw
 29  * @override
 30  */
 31 rock.game.GameEngineWindow.prototype.draw = function (graphicsEngine) {
 32     var gameState = this.gameEngine.update();
 33     this.drawUI(graphicsEngine);
 34     this.gameDirector.handleGameState(gameState);
 35 };
 36 
 37 rock.game.GameEngineWindow.prototype.start = function () {
 38     rock.timer.registerTimerFunction(this, this.redraw, []);
 39     rock.timer.start();
 40 };
 41 
 42 rock.game.GameEngineWindow.prototype.stop = function () {
 43     rock.timer.unregisterTimerFunction(this.redraw);
 44     rock.timer.stop();
 45 };
