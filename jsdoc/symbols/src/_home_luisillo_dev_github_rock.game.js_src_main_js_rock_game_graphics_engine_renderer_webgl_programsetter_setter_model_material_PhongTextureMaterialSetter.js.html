 48 rock.game.graphics.engine.renderer.webgl.programsetter.setter.model.material.PhongTextureMaterialSetter.prototype.setAttributesFromMaterial = function (program, material) {
 49     var ambientTexture = material.getAmbientTexture();
 50     var diffuseTexture = material.getDiffuseTexture();
 51     var specularTexture = material.getSpecularTexture();
 52 
 53     if (ambientTexture != null) {
 54         program.populateTexture(this.TEXTURE_AMBIENT, ambientTexture.getImage());
 55     }
 56 
 57     if (diffuseTexture != null) {
 58         program.populateTexture(this.TEXTURE_DIFFUSE, diffuseTexture.getImage());
 59     }
 60 
 61     if (specularTexture != null) {
 62         program.populateTexture(this.TEXTURE_SPECULAR, specularTexture.getImage());
 63     }
 64 };
