 13 rock.game.graphics.model.JSONModelUtils.createTextureModelGroup = function (JSONModelList, repository) {
 14     var groupModel = new rock.game.graphics.model.GroupModel();
 15 
 16     var i, JSONModel;
 17     for (i = 0; i < JSONModelList.length; i++) {
 18         JSONModel = JSONModelList[i];
 19         groupModel.addModel(this.createTextureModel(JSONModel, repository));
 20     }
 21 
 22     return groupModel;
 23 };

111 rock.game.graphics.model.JSONModelUtils.createWireframeColorModelFromMesh = function (wireframeMesh) {
112     var wireframeColorModel = new rock.game.graphics.model.WireframeColorModel();
113     wireframeColorModel.setMesh(wireframeMesh);
114 
115     var material = new rock.game.graphics.model.material.ColorMaterial();
116     var color = new rock.graphics.Color(255, 0, 0);
117     material.setColor(color);
118     wireframeColorModel.setMaterial(material);
119 
120     return wireframeColorModel;
121 };
