 52 /**
 53  * Default implementation. You can override and apply any filter, customization,.. you want
 54  *
 55  * @param lighting
 56  * @param {rock.game.graphics.scene.Camera} camera
 57  * @param {rock.js.Array} representations
 58  *
 59  */
 60 rock.game.graphics.engine.renderer.RendererContext.prototype.render = function (lighting, camera, representations) {
 61     var i, representation;
 62     this.setUp();
 63     this.clear();
 64     camera.updateMatrices();
 65     var length = representations.getLength();
 66     for (i = 0; i < length; i++) {
 67         representation = representations.getValue(i);
 68         this.renderRepresentation(lighting, camera, representation);
 69     }
 70     this.tearDown();
 71 };
 72 
 73 rock.game.graphics.engine.renderer.RendererContext.prototype.renderRepresentation = function (lighting, camera, representation) {
 74     representation.update();
 75     var renderables = representation.getRenderables();
 76     var length = renderables.getLength();
 77     var i;
 78     for (i = 0; i < length; i++) {
 79         this.renderRenderable(lighting, camera, renderables.getValue(i));
 80     }
 81 };
 82 
 83 rock.game.graphics.engine.renderer.RendererContext.prototype.renderRenderable = function (lighting, camera, renderable) {
 84     var renderer = this.getRenderer(lighting.getType(), renderable.getModelType());
 85     renderer.setUp();
 86     renderer.render(lighting, camera, renderable);
 87     renderer.tearDown();
 88 };
 89 
 90 rock.game.graphics.engine.renderer.RendererContext.prototype.getRenderer = function (lightingType, modelType) {
 91     var renderers = this.renderers;
 92     var i, renderer;
 93     for (i = 0; i < renderers.length; i++) {
 94         renderer = renderers[i];
 95         if (renderer.getLightingType() == lightingType && renderer.getModelType() == modelType) {
 96             return renderer;
 97         }
 98     }
 99 
100     // Trick. If no renderer found, we try to get one that doesn't need lighting
101     for (i = 0; i < renderers.length; i++) {
102         renderer = renderers[i];
103         if (renderer.getLightingType() == rock.game.constants.NO_LIGHTING_NEEDED_TYPE
104             && renderer.getModelType() == modelType) {
105             return renderer;
106         }
107     }
108     return null;
109 };
