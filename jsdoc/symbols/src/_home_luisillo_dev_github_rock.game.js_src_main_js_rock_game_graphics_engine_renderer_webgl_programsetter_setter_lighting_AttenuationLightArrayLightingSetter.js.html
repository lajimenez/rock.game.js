  4  * Setter for {@link rock.game.graphics.scene.LightArrayLighting}. Lights may have attenuation and

  6  *

  8  *         the WebGL context
  9  * @param varName

 11  * @param maxLights

 19 rock.game.graphics.engine.renderer.webgl.programsetter.setter.lighting.AttenuationLightArrayLightingSetter = function (glContext, varName, maxLights) {
 20     this.UNIFORM_LIGHT_CONSTANT_ATTENUATION = varName + '[X].constantAttenuation';
 21     this.UNIFORM_LIGHT_LINEAR_ATTENUATION = varName + '[X].linearAttenuation';
 22     this.UNIFORM_LIGHT_EXPONENTIAL_ATTENUATION = varName + '[X].exponentialAttenuation';
 23 
 24     rock.super_(this, arguments);
 25 };

 31 rock.game.graphics.engine.renderer.webgl.programsetter.setter.lighting.AttenuationLightArrayLightingSetter.prototype.getCustomUniformNames = function (position) {

 37 rock.game.graphics.engine.renderer.webgl.programsetter.setter.lighting.AttenuationLightArrayLightingSetter.prototype.getAttenuationLightUniformNames = function (position) {
 38     var uniforms = [];
 39     uniforms.push(this.getUniformNameInPosition(this.UNIFORM_LIGHT_CONSTANT_ATTENUATION, position));
 40     uniforms.push(this.getUniformNameInPosition(this.UNIFORM_LIGHT_LINEAR_ATTENUATION, position));
 41     uniforms.push(this.getUniformNameInPosition(this.UNIFORM_LIGHT_EXPONENTIAL_ATTENUATION, position));
 42     return uniforms;
 43 };

 50 rock.game.graphics.engine.renderer.webgl.programsetter.setter.lighting.AttenuationLightArrayLightingSetter.prototype.setAttenuationLightUniforms = function (program, lighting, position) {
 51     var gl = this.glContext.getHTMLContext();
 52     var light = lighting.getLight(position);
 53 
 54     gl.uniform1f(
 55         program.getUniformLocation(this.getUniformNameInPosition(this.UNIFORM_LIGHT_CONSTANT_ATTENUATION, position)),
 56         light.getConstantAttenuation()
 57     );
 58 
 59     gl.uniform1f(
 60         program.getUniformLocation(this.getUniformNameInPosition(this.UNIFORM_LIGHT_LINEAR_ATTENUATION, position)),
 61         light.getLinearAttenuation()
 62     );
 63 
 64     gl.uniform1f(
 65         program.getUniformLocation(this.getUniformNameInPosition(this.UNIFORM_LIGHT_EXPONENTIAL_ATTENUATION, position)),
 66         light.getExponentialAttenuation()
 67     );
 68 };
