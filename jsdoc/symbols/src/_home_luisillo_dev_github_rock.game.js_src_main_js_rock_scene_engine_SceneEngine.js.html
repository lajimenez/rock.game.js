  3 /**
  4  * Game engine for scene.
  5  *
  6  * @param {rock.graphics.engine.WebGLGraphicsEngine} graphicsEngine
  7  *
  8  * @param {rock.game.Repository} repository
  9  *

 11  * @extends rock.game.engine.GameEngine
 12  *
 13  * @author Luis Alberto Jim√©nez
 14  */
 15 rock.scene.engine.SceneEngine = function (graphicsEngine, repository) {
 16     var renderEngine = new rock.game.graphics.engine.webgl.RockBasicRenderEngine(graphicsEngine);
 17     var representationAttendant = new rock.scene.engine.RepresentationAttendant(renderEngine, repository);
 18     rock.super_(this, [renderEngine, representationAttendant]);
 19 };
 20 
 21 rock.extends_(rock.scene.engine.SceneEngine, rock.game.engine.GameEngine);
 22 
 23 /**
 24  * @see rock.game.engine.GameEngine#runLogic
 25  * @override
 26  */
 27 rock.scene.engine.SceneEngine.prototype.runLogic = function () {
 28     var scene = this.world;
 29     this.lighting = scene.getLightingNode().getLighting();
 30     this.camera = scene.getCameraNode().getCamera();
 31     this.renderEngine.setClearColor(scene.getBackgroundColor());
 32     // result is not used in scene...
 33     return null;
 34 };

 36 /**
 37  * Create a node for add an object to the scene
 38  *
 39  * IMPORTANT: This function creates adapters that you should release manually.
 40  * So when you are not gonna use a ModelNode anymore, you should call 'releaseResources()' method on it.
 41  *
 42  * @param {rock.game.graphics.model.Model} model
 43  *
 44  * @returns {rock.scene.node.ModelNode}
 45  */
 46 rock.scene.engine.SceneEngine.prototype.createModelNode = function (model) {
 47     var modelNode = new rock.scene.node.ModelNode(this.world);
 48     var representation = this.createRepresentation(modelNode);
 49     representation.createRenderablesFromModel(model);
 50 
 51     modelNode.setRepresentation(representation);
 52     return modelNode;
 53 };

 55 /**
 56  * Create a node for set lighting to the scene
 57  *
 58  * @param {rock.game.graphics.scene.Lighting} lighting
 59  *
 60  * @returns {rock.scene.node.LightingNode}
 61  */
 62 rock.scene.engine.SceneEngine.prototype.createLightingNode = function (lighting) {
 63     var lightNode = new rock.scene.node.LightingNode();
 64     lightNode.setLighting(lighting);
 65     return lightNode;
 66 };

 68 /**

 71  * @param {rock.game.graphics.scene.Camera} camera

 75 rock.scene.engine.SceneEngine.prototype.createCameraNode = function (camera) {
