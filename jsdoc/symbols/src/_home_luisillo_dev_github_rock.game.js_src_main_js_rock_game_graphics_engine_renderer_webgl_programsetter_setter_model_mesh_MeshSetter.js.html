 41 rock.game.graphics.engine.renderer.webgl.programsetter.setter.model.mesh.MeshSetter.prototype.setMeshAttributes = function (program, mesh, meshAdapter) {

 49 rock.game.graphics.engine.renderer.webgl.programsetter.setter.model.mesh.MeshSetter.prototype.setAttributesFromMesh = function (program, mesh) {
 50     var vertices = mesh.getVertices();
 51     var indexes = mesh.getIndexes();
 52 
 53     program.populateAttribute(this.ATTRIBUTE_VERTEX, new Float32Array(vertices), 3);
 54     program.populateIndexes(new Uint16Array(indexes));
 55 };

 57 rock.game.graphics.engine.renderer.webgl.programsetter.setter.model.mesh.MeshSetter.prototype.setAttributesFromMeshAdapter = function (program, meshAdapter) {
 58     var verticesAdapter = meshAdapter.getVerticesAdapter();
 59     var indexesAdapter = meshAdapter.getIndexesAdapter();
 60 
 61     program.bufferAttribute(this.ATTRIBUTE_VERTEX, verticesAdapter.getBuffer(), 3);
 62     program.bufferIndexes(indexesAdapter.getBuffer());
 63 };
