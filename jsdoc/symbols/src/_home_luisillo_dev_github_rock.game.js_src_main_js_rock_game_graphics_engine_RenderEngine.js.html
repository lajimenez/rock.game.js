  3 /**
  4  * The render engine is responsible for rendering game. Internally, this class will use a
  5  * {@link rock.game.graphics.engine.renderer.RendererContext} to do the render
  6  *
  7  * @param {rock.graphics.IGraphicsEngine} graphicsEngine
  8  *
  9  * @constructor
 10  * @abstract
 11  *
 12  * @author Luis Alberto Jim√©nez
 13  */
 14 rock.game.graphics.engine.RenderEngine = function (graphicsEngine) {
 15     this.graphicsEngine = graphicsEngine;
 16     this.rendererContexts = {};
 17     this.currentRendererContext = null;
 18 
 19     this.renderableConstructors = {};
 20 };
 21 
 22 /**
 23  * Add a new renderable constructor that will be used for a specific model type.
 24  *
 25  * @param modelType
 26  * @param renderableConstructor
 27  */
 28 rock.game.graphics.engine.RenderEngine.prototype.addRenderableConstructor = function (modelType, renderableConstructor) {
 29     this.renderableConstructors[modelType] = renderableConstructor;
 30 };
 31 
 32 /**
 33  * Add a new renderer context
 34  *
 35  * @param {rock.game.graphics.engine.renderer.RendererContext} rendererContext
 36  */
 37 rock.game.graphics.engine.RenderEngine.prototype.addRendererContext = function (rendererContext) {
 38     this.rendererContexts[rendererContext.getId()] = rendererContext;
 39 };
 40 
 41 /**
 42  * Set renderer context to use
 43  *
 44  * @param {String} id
 45  */
 46 rock.game.graphics.engine.RenderEngine.prototype.setActiveRendererContext = function (id) {
 47     this.currentRendererContext = this.rendererContexts[id];
 48 };
 49 
 50 /**
 51  * Do the render
 52  *
 53  * @param {rock.game.graphics.scene.Camera} camera
 54  * @param {rock.game.graphics.scene.Lighting} lighting
 55  * @param {rock.js.Array} representations
 56  *      array of {@link rock.game.graphics.scene.Representation}
 57  */
 58 rock.game.graphics.engine.RenderEngine.prototype.render = function (camera, lighting, representations) {
 59    this.currentRendererContext.render(camera, lighting, representations);
 60 };
 61 
 62 /**
 63  * Get the graphicsEngine
 64  */
 65 rock.game.graphics.engine.RenderEngine.prototype.getGraphicsEngine = function() {
 66     return this.graphicsEngine;
 67 };
 68 
 69 /**
 70  * Set the color used for clear window before drawing
 71  *
 72  * @param {rock.graphics.Color} color
 73  */
 74 rock.game.graphics.engine.RenderEngine.prototype.setClearColor = function (color) {
 75     var rendererContexts = this.rendererContexts;
 76     var propertyName, rendererContext;
 77 
 78     for (propertyName in rendererContexts) {
 79         rendererContext = rendererContexts[propertyName];
 80         if (rock.instanceof_(rendererContext, rock.game.graphics.engine.renderer.RendererContext)) {
 81             rendererContext.setClearColor(color);
 82         }
 83     }
 84 };
 85 
 86 /**
 87  * Create a renderable for an specific model. This method should only be call for the
 88  * {@link rock.game.engine.RepresentationAttendant}.
 89  *
 90  * @param {rock.game.graphics.model.Model} model
 91  *
 92  * @returns {rock.game.graphics.engine.renderer.ModelRenderable} renderable
 93  */
 94 rock.game.graphics.engine.RenderEngine.prototype.createRenderable = function (model) {
 95     var type = model.getType();
 96 
 97     var renderableConstructor = this.renderableConstructors[type];
 98     if (rock.util.JsUtils.isNullOrUndefined(renderableConstructor)) {
 99         return null;
100     }
101     var renderable = new renderableConstructor();
102     renderable.setModel(model);
103     renderable.setModelType(model.getType());
104     return renderable;
105 };
