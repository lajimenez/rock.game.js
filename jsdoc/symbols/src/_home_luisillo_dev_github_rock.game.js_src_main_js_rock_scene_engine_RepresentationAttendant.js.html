 15 rock.scene.engine.RepresentationAttendant = function (renderEngine, repository) {
 16     rock.super_(this, arguments);
 17     this.representations_ma = new rock.js.Array();
 18 };
 19 
 20 rock.extends_(rock.scene.engine.RepresentationAttendant, rock.game.engine.RepresentationAttendant);
 21 
 22 /**
 23  * @see rock.game.engine.RepresentationAttendant#createRepresentation
 24  * @override
 25  */
 26 rock.scene.engine.RepresentationAttendant.prototype.createRepresentation = function (object) {
 27     // In this case, the switch could be avoided...
 28     var representation = null;
 29     var type = object.getType();
 30     switch (type) {
 31         case rock.scene.constants.MODEL_NODE_TYPE:
 32             representation = new rock.scene.graphics.scene.ModelNodeRepresentation(this, object);
 33             break;
 34         default:
 35             // Throw an exception???
 36     }
 37 
 38     return representation;
 39 };
 40 
 41 /**
 42  * @see rock.game.engine.RepresentationAttendant#getRepresentations
 43  * @override
 44  */
 45 rock.scene.engine.RepresentationAttendant.prototype.getRepresentations = function (scene) {
 46     var modelNodes = scene.getModelNodes();
 47     var representations = this.representations_ma;
 48     var i, modelNode;
 49 
 50     representations.clear();
 51     for (i = 0; i < modelNodes.length; i++) {
 52         modelNode = modelNodes[i];
 53         representations.addValue(modelNode.getRepresentation())
 54     }
 55 
 56     return representations;
 57 };
