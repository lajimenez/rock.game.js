 15 rock.game.engine.GameEngine = function (renderEngine, representationAttendant) {
 16     this.window = null;
 17     this.renderEngine = renderEngine;
 18     this.representationAttendant = representationAttendant;
 19 
 20     this.lighting = null;
 21     this.camera = null;
 22     this.world = null;

 91 /**
 92  * Run game logic. It must return a game state object with all relevant events that have happened.
 93  *
 94  * @returns {rock.game.engine.GameState} the result of running logic
 95  *
 96  * @function
 97  */
 98 rock.game.engine.GameEngine.prototype.runLogic = rock.abstract_;
 99 
100 /**
101  * Called in the game loop.
102  *
103  * @return {rock.game.engine.GameState} the result of running logic
104  */
105 rock.game.engine.GameEngine.prototype.update = function () {
106     var logicExecutionResult = this.runLogic();
107     this.render();
108     return logicExecutionResult;
109 };
110 
111 rock.game.engine.GameEngine.prototype.render = function () {
112     var representations = this.representationAttendant.getRepresentations(this.world);
113     this.renderEngine.render(this.lighting, this.camera, representations);
114 };
