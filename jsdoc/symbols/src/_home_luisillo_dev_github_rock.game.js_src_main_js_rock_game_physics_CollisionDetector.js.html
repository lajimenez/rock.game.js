 10 rock.game.physics.CollisionDetector = function () {

 12 };
 13 
 14 /**
 15  * Add a barrier
 16  *
 17  * @param {rock.game.physics.Barrier} barrier
 18  */
 19 rock.game.physics.CollisionDetector.prototype.addBarrier = function (barrier) {
 20     this.barriers.addValue(barrier);
 21 };
 22 
 23 /**
 24  * Check if the point cause some collision
 25  *
 26  * @param {rock.geometry.Point3} point
 27  * @return {Boolean}
 28  */
 29 rock.game.physics.CollisionDetector.prototype.intersectWithPoint = function (point) {
 30     var barriers = this.barriers;
 31     var length = barriers.getLength();
 32     var i, barrier;
 33 
 34     for (i = 0; i < length; i++) {
 35         barrier = barriers.getValue(i);
 36         if (barrier.intersectWithPoint(point)) {
 37             return true;
 38         }
 39     }
 40 
 41     return false;
 42 };
