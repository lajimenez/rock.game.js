 46 rock.game.graphics.engine.renderer.webgl.programsetter.setter.lighting.SpotlightLightingSetter.prototype.setSpotlightLightUniforms = function (program, lighting, position, camera) {
 47     var gl = this.glContext.getHTMLContext();
 48     var light = lighting.getLight(position);
 49 
 50     gl.uniform1f(
 51         program.getUniformLocation(this.getUniformNameInPosition(this.UNIFORM_LIGHT_CONE_ANGLE, position)),
 52         light.getConeAngle()
 53     );
 54 
 55     var point = this.point3_ma;
 56     var coneDirection = this.vec3_ma;
 57     var lightConeDirection = light.getConeDirection();
 58     point.setX(lightConeDirection.getX());
 59     point.setY(lightConeDirection.getY());
 60     point.setZ(lightConeDirection.getZ());
 61     var coneDirectionPoint = this.getPointInCorrectSpace(point,
 62         camera.getLookAtViewMatrix(), light.getIsPositionInCameraSpace(), true);
 63 
 64     coneDirection.setX(coneDirectionPoint.getX());
 65     coneDirection.setY(coneDirectionPoint.getY());
 66     coneDirection.setZ(coneDirectionPoint.getZ());
 67     coneDirection.normalize();
 68     gl.uniform3f(
 69         program.getUniformLocation(this.getUniformNameInPosition(this.UNIFORM_LIGHT_CONE_DIRECTION, position)),
 70         coneDirection.getX(), coneDirection.getY(), coneDirection.getZ()
 71     );
 72 };
