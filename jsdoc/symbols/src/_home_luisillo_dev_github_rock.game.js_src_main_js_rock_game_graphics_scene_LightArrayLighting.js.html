  3 /**
  4  * Lighting model composed for multiple lights.
  5  *
  6  * @constructor
  7  * @extends rock.game.graphics.scene.Lighting
  8  *
  9  * @author Luis Alberto Jim√©nez
 10  */
 11 rock.game.graphics.scene.LightArrayLighting = function (maxLights, type) {
 12     rock.super_(this, [type]);
 13     this.maxLights = rock.game.graphics.scene.LightArrayLighting.DEFAULT_MAX_LIGHTS;
 14     if (!rock.util.JsUtils.isNullOrUndefined(maxLights)) {
 15         this.maxLights = maxLights;
 16     }
 17 
 18     this.lights = new rock.js.Array();
 19 };
 20 
 21 rock.extends_(rock.game.graphics.scene.LightArrayLighting, rock.game.graphics.scene.Lighting);
 22 
 23 rock.game.graphics.scene.LightArrayLighting.DEFAULT_MAX_LIGHTS = 10;
 24 
 25 /**
 26  * Get the directional light at some position
 27  *
 28  * @param position
 29  *      the position of the light
 30  *
 31  * @return {rock.game.graphics.lighting.Light} the light
 32  */
 33 rock.game.graphics.scene.LightArrayLighting.prototype.getLight = function(position) {
 34     return this.lights.getValue(position);
 35 };
 36 
 37 /**
 38  * Add a new directional light
 39  *
 40  * @param {rock.game.graphics.lighting.Light} light
 41  *          the light
 42  */
 43 rock.game.graphics.scene.LightArrayLighting.prototype.addLight = function(light) {
 44     var lights = this.lights;
 45     if (lights.getLength() >= this.maxLights) {
 46         // TODO: throw exception here
 47         return;
 48     }
 49     this.lights.addValue(light);
 50 };
 51 
 52 /**
 53  * Remove the directional light
 54  *
 55  * @param {rock.game.graphics.lighting.Light} light
 56  *          the light
 57  */
 58 rock.game.graphics.scene.LightArrayLighting.prototype.removeLight = function(light) {
 59     this.lights.removeValue(light);
 60 };
 61 
 62 /**
 63  * Clear all lights
 64  */
 65 rock.game.graphics.scene.LightArrayLighting.prototype.clearLights = function () {
 66     this.lights.clear();
 67 };
 68 
 69 rock.game.graphics.scene.LightArrayLighting.prototype.getLightsCount = function () {
 70     return this.lights.getLength();
 71 };
