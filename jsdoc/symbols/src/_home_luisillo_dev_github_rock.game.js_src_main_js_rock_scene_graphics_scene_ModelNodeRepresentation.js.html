 30 rock.scene.graphics.scene.ModelNodeRepresentation.prototype.createRenderablesFromModel = function (baseModel) {
 31     this.model = baseModel;
 32     var renderables = this.renderables;
 33     var modelArray = this.getAsModelArray(baseModel);
 34     var renderEngine = this.representationAttendant.getRenderEngine();
 35 
 36     var i, model, renderable, adapter;
 37     for (i = 0; i < modelArray.length; i++) {
 38         model = modelArray[i];
 39         renderable = renderEngine.createRenderable(model);
 40         adapter = renderEngine.createAdapter(model);
 41         adapter.build();
 42         renderable.setModelAdapter(adapter);
 43         renderables.addValue(renderable);
 44     }
 45 };
 46 
 47 rock.scene.graphics.scene.ModelNodeRepresentation.prototype.releaseRenderables = function () {
 48     var renderables = this.renderables;
 49     var length = renderables.getLength();
 50 
 51     var i, renderable, adapter;
 52     for (i = 0; i < length; i++) {
 53         renderable = renderables.getValue(i);
 54         adapter = renderable.getModelAdapter();
 55         if (adapter != null) {
 56             adapter.release();
 57         }
 58     }
 59 
 60     renderables.clear(true);
 61 };

 63 rock.scene.graphics.scene.ModelNodeRepresentation.prototype.getAsModelArray = function (model) {
 64     var modelArray = null;
 65 
 66     if (rock.instanceof_(model, rock.game.graphics.model.GroupModel)) {
 67         return model.getModels();
 68     } else {
 69         modelArray = [];
 70         modelArray.push(model);
 71     }
 72 
 73     return modelArray;
 74 };
