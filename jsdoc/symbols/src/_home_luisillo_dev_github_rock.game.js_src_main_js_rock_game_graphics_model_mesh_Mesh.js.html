 46     this.setVertices(JSON.vertices);

 49     var BBOX = new rock.geometry.BBOX(BBOXValues[0], BBOXValues[1],

118 /**
119  * Convert the index to represent lines instead of faces (so it can be drawn with gl.LINES)
120  */
121 rock.game.graphics.model.mesh.Mesh.prototype.getAsWireframeMesh = function () {
122     var clone = this.clone();
123 
124     var indexes = this.indexes;
125     var length = indexes.length;
126     var newIndexes = [];
127     var i, v1, v2, v3;
128 
129     for (i = 0; i < length; i = i + 3){
130         v1 = indexes[i];
131         v2 = indexes[i + 1];
132         v3 = indexes[i + 2];
133 
134         newIndexes.push(v1);
135         newIndexes.push(v2);
136 
137         newIndexes.push(v2);
138         newIndexes.push(v3);
139 
140         newIndexes.push(v3);
141         newIndexes.push(v1);
142     }
143 
144     clone.setIndexes(newIndexes);
145     return clone;
146 };

157 rock.game.graphics.model.mesh.Mesh.createMeshFromBBOX = function (BBOX) {
158     var bboxMesh = new rock.game.graphics.model.mesh.Mesh();
159 
160     var xMin = BBOX.getXMin();
161     var xMax = BBOX.getXMax();
162     var yMin = BBOX.getYMin();
163     var yMax = BBOX.getYMax();
164     var zMin = BBOX.getZMin();
165     var zMax = BBOX.getZMax();
166 
167     var vertices = [];
168     vertices.push(xMin, yMin, zMin);
169     vertices.push(xMin, yMin, zMax);
170     vertices.push(xMax, yMin, zMax);
171     vertices.push(xMax, yMin, zMin);
172 
173     vertices.push(xMin, yMax, zMin);
174     vertices.push(xMin, yMax, zMax);
175     vertices.push(xMax, yMax, zMax);
176     vertices.push(xMax, yMax, zMin);
177 
178     var indexes = [];
179     indexes.push(0, 1);
180     indexes.push(1, 2);
181     indexes.push(2, 3);
182     indexes.push(3, 0);
183 
184     indexes.push(4, 5);
185     indexes.push(5, 6);
186     indexes.push(6, 7);
187     indexes.push(7, 4);
188 
189     indexes.push(0, 4);
190     indexes.push(1, 5);
191     indexes.push(2, 6);
192     indexes.push(3, 7);
193 
194     bboxMesh.setVertices(vertices);
195     bboxMesh.setIndexes(indexes);
196 
197     bboxMesh.setBBOX(BBOX.clone());
198 
199     return bboxMesh;
200 };

202 /**
203  * Compute the BBOX of the mesh
204  */
205 rock.game.graphics.model.mesh.Mesh.prototype.computeBBOX = function () {
206     var vertices = this.vertices;
207     var length = vertices.length;
208 
209     if (length == 0) {
210         return;
211     }
212 
213     var xMin = vertices[0];
214     var xMax = vertices[0];
215     var yMin = vertices[1];
216     var yMax = vertices[1];
217     var zMin = vertices[2];
218     var zMax = vertices[2];
219     var i, x, y, z;
220 
221     for (i = 3; i < length; i = i + 3){
222         x = vertices[i];
223         y = vertices[i +1];
224         z = vertices[i + 2];
225         if (x < xMin) {
226             xMin = x;
227         }
228         if (x > xMax) {
229             xMax = x;
230         }
231 
232         if (y < yMin) {
233             yMin = y;
234         }
235         if (y > yMax) {
236             yMax = y;
237         }
238 
239         if (z < zMin) {
240             zMin = z;
241         }
242         if (z > zMax) {
243             zMax = z;
244         }
245     }
246 
247     this.BBOX = new rock.geometry.BBOX(xMin, xMax, yMin, yMax, zMin, zMax);
248 };
