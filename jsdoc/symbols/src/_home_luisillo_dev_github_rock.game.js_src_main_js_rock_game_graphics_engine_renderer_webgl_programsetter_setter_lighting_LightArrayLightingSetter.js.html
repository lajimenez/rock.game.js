 57 rock.game.graphics.engine.renderer.webgl.programsetter.setter.lighting.LightArrayLightingSetter.prototype.setValues = function (program, settable) {
 58     var lighting = settable.getLighting();
 59     var camera = settable.getCamera();
 60     var lightsCount = lighting.getLightsCount();
 61     var i;
 62     for (i = 0; i < lightsCount; i++) {
 63         this.setCustomLightUniforms(program, lighting, i, camera);
 64     }
 65 
 66     var gl = this.glContext.getHTMLContext();
 67     gl.uniform1i(program.getUniformLocation(this.UNIFORM_LIGHT_COUNT), lightsCount);
 68 };

 74 rock.game.graphics.engine.renderer.webgl.programsetter.setter.lighting.LightArrayLightingSetter.prototype.setLightUniforms = function (program, lighting, position, camera) {
 75     var gl = this.glContext.getHTMLContext();
 76     var light = lighting.getLight(position);
 77 
 78     var lightPosition = this.getPointInCorrectSpace(light.getPosition(), camera.getLookAtViewMatrix(),
 79         light.getIsPositionInCameraSpace(), light.getIsPositionAVector());
 80     gl.uniform3f(
 81         program.getUniformLocation(this.getUniformNameInPosition(this.UNIFORM_LIGHT_POSITION, position)),
 82         lightPosition.getX(), lightPosition.getY(), lightPosition.getZ()
 83     );
 84 
 85     var ambient = light.getAmbient();
 86     gl.uniform3f(
 87         program.getUniformLocation(this.getUniformNameInPosition(this.UNIFORM_LIGHT_AMBIENT, position)),
 88         ambient.getNormalizedRed(), ambient.getNormalizedGreen(), ambient.getNormalizedBlue()
 89     );
 90 
 91     var diffuse = light.getDiffuse();
 92     gl.uniform3f(
 93         program.getUniformLocation(this.getUniformNameInPosition(this.UNIFORM_LIGHT_DIFFUSE, position)),
 94         diffuse.getNormalizedRed(), diffuse.getNormalizedGreen(), diffuse.getNormalizedBlue()
 95     );
 96 
 97     var specular = light.getSpecular();
 98     gl.uniform3f(
 99         program.getUniformLocation(this.getUniformNameInPosition(this.UNIFORM_LIGHT_SPECULAR, position)),
100         specular.getNormalizedRed(), specular.getNormalizedGreen(), specular.getNormalizedBlue()
101     );
102 };
